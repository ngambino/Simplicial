\end{prooftree}

\justifies

\fbox{\begin{minipage}{14cm}  \smallskip
\[
\begin{prooftree}
\Gamma \vdash A \co \type \qquad
\Gamma, x \co A \vdash B \co \type
\justifies
\Gamma \vdash (\Pi x \co A) B \co \type
\end{prooftree} 
\] 
\medskip
\[
\begin{gathered}
\begin{prooftree}
\Gamma, x \co A \vdash b \co B
\justifies
\Gamma\vdash (\lambda x \co A) b \co (\Pi x \co A) B
\end{prooftree}
\end{gathered} \qquad
\begin{gathered}
\begin{prooftree}
\Gamma \vdash f \co (\Pi x \co A) B \quad
\Gamma \vdash a \co A 
\justifies
\Gamma \vdash \mathsf{app}(f,a) \co B[a/x]
\end{prooftree}
\end{gathered}
\]
\medskip
\[
\begin{gathered}
\begin{prooftree}
\Gamma, x \co A \vdash b \co B \quad
\Gamma \vdash a \co A 
\justifies
\Gamma \vdash \mathsf{app}((\lambda x \co A) b, a) = b[a/x] \co B[a/x]
\end{prooftree}
\end{gathered}
\qquad
\begin{gathered}
\begin{prooftree}
\Gamma \vdash f \co (\Pi x \co A) B \quad
\justifies
\Gamma \vdash \eta_f \co \Id(f \, , (\lambda x \co A) \mathsf{app}(f, x) )
\end{prooftree}
\end{gathered}
\] \medskip
\end{minipage} }
\medskip
\caption{Rules for $\Pi$-types.}

\end{table}

\justifies
\Gamma \vdash \mathsf{J}(a,a,\mathsf{refl}(a), e) = e(a) \co E(a,a,\mathsf{refl}(a))
